<script>
function simular() {
    const monto = parseFloat(document.getElementById('monto').value);
    const plazo = parseInt(document.getElementById('plazo').value);
    const tasaAnual = 0.156;
    const seguro = 0.00093;

    let valido = true;

    if (isNaN(monto) || monto <= 0) {
        document.getElementById('montoError').style.display = 'block';
        valido = false;
    } else {
        document.getElementById('montoError').style.display = 'none';
    }

    if (isNaN(plazo) || plazo <= 0) {
        document.getElementById('plazoError').style.display = 'block';
        valido = false;
    } else {
        document.getElementById('plazoError').style.display = 'none';
    }

    if (!valido) return;

    const tasaMensual = tasaAnual / 12;
    const cuota = (monto * tasaMensual) / (1 - Math.pow(1 + tasaMensual, -plazo));

    let saldo = monto;
    const tbody = document.querySelector('#tablaResultados tbody');
    tbody.innerHTML = '';

    for (let i = 1; i <= plazo; i++) {
        const interes = saldo * tasaMensual;
        const capital = cuota - interes;
        const seguroCuota = monto * seguro;
        saldo -= capital;

        tbody.innerHTML += `
            <tr>
                <td>${i}</td>
                <td>${saldo.toFixed(2)}</td>
                <td>${capital.toFixed(2)}</td>
                <td>${interes.toFixed(2)}</td>
                <td>${seguroCuota.toFixed(2)}</td>
                <td>${(cuota + seguroCuota).toFixed(2)}</td>
            </tr>`;
    }

    document.getElementById('tablaResultados').style.display = 'table';
}
</script>
